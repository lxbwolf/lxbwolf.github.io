func merge(intervals [][]int) [][]int {

	for i := 0; i < len(intervals); {
		merged := false
		for j := i + 1; j < len(intervals); j++ {
			x, y := intervals[i], intervals[j]
			
			if min(x[1], y[1]) >= max(x[0], y[0]) {
				merged = true

	}
