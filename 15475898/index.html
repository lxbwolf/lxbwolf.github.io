	private Config config;
	/**

		//getDB

	/**

		config.setUserPw(pw);
		//updateDB
	}
	
	/**
	 * é…ç½®æ–‡ä»¶åŒæ­¥åˆ°zookeeper

	public void syncConfigToZk(){

		}
		zk.writeData("/zkConfig", config);
		zk.close();
	}
}
